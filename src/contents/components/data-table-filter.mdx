> This component is a customized version of the **Data Table Filter** from [bazza/ui](https://bazzaui.com). It has been adapted to use the project's existing UI primitives and adds an Indonesian (`id`) locale for date formatting.

`DataTableFilter` provides a composable filter bar for TanStack Table. It supports `text`, `option`, `multiOption`, `number`, `date`, and `boolean` column types, with both client-side and server-side filter strategies.

## Installation

Add this component into your project using Shadcn CLI. It has no external dependencies beyond `@tanstack/react-table`, `date-fns`, and `react-day-picker`.

## Usage

```tsx
"use client"

import { useDataTableFilters, DataTableFilter } from "@/blocks/data-table-filter/data-table-filter"
import { TextIcon, CircleIcon } from "lucide-react"

type Row = { name: string; status: "active" | "inactive" | "pending" }

const data: Row[] = [
  { name: "Invoice #001", status: "active" },
  { name: "Invoice #002", status: "pending" },
  { name: "Invoice #003", status: "inactive" },
]

const columnsConfig = [
  {
    id: "name",
    accessor: (r: Row) => r.name,
    displayName: "Name",
    icon: TextIcon,
    type: "text",
  },
  {
    id: "status",
    accessor: (r: Row) => r.status,
    displayName: "Status",
    icon: CircleIcon,
    type: "option",
    options: [
      { label: "Active", value: "active" },
      { label: "Inactive", value: "inactive" },
      { label: "Pending", value: "pending" },
    ],
  },
] as const

export function MyFilter() {
  const { columns, filters, actions } = useDataTableFilters({
    strategy: "client",
    data,
    columnsConfig,
  })

  return <DataTableFilter columns={columns} filters={filters} actions={actions} />
}
```

Pass `strategy: "server"` and handle `filters` externally via `onFiltersChange` for server-side filtering.
